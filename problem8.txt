import groovy.json.JsonSlurperClassic
pipeline {
    agent any
    
    stages {
        stage('fetch_data') {
            steps {
                script {
                    def fetch = new URL("https://jsonplaceholder.typicode.com/todos").openConnection();
                    fetch.requestMethod = "GET"
                    def array = fetch.getInputStream().getText()
                    def data_in_json = new JsonSlurperClassic().parseText(array)
                    for(element in data_in_json){
                       if(element.completed == false){
                            println(element);
                       } 
                    }
                }
            }
        }
    }
}
